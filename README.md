from IPython.display import clear_output

def exibir_tabuleiro(tabuleiro):
    """Imprime o tabuleiro com cores e organização."""
    print("\n")
    print(f"  {tabuleiro[0]}  |  {tabuleiro[1]}  |  {tabuleiro[2]} ")
    print("-----|-----|-----")
    print(f"  {tabuleiro[3]}  |  {tabuleiro[4]}  |  {tabuleiro[5]} ")
    print("-----|-----|-----")
    print(f"  {tabuleiro[6]}  |  {tabuleiro[7]}  |  {tabuleiro[8]} ")
    print("\n")

def verificar_vitoria(tabuleiro, jogador_colorido):
    condicoes = [
        [0, 1, 2], [3, 4, 5], [6, 7, 8],  # Linhas
        [0, 3, 6], [1, 4, 7], [2, 5, 8],  # Colunas
        [0, 4, 8], [2, 4, 6]              # Diagonais
    ]
    for condicao in condicoes:
        if tabuleiro[condicao[0]] == tabuleiro[condicao[1]] == tabuleiro[condicao[2]] == jogador_colorido:
            return True
    return False

def jogo_da_velha():
    tabuleiro = [str(i) for i in range(1, 10)]

    # Dicionário para mapear a letra pura para a sua versão colorida
    cores = {
        "X": "\033[1;34mX\033[0m",  # Azul
        "O": "\033[1;31mO\033[0m"   # Vermelho
    }

    # O jogador_atual agora é apenas o texto puro "X" ou "O"
    jogador_atual = "X"
    jogadas = 0

    while jogadas < 9:
        clear_output()
        print("=" * 35)
        print("      🎮 JOGO DA VELHA 🎮      ")
        print("=" * 35)

        exibir_tabuleiro(tabuleiro)

        try:
            # Agora aparece limpo: "Jogador X, escolha uma posição..."
            escolha = int(input(f"Jogador {jogador_atual}, escolha uma posição (1-9): ")) - 1
        except ValueError:
            print("❌ Por favor, digite um número válido!")
            input("Aperte Enter para continuar...") # Espera o usuário em vez de pausar por tempo
            continue
        if escolha < 0 or escolha > 8 or tabuleiro[escolha] != str(escolha + 1):
            print("⚠️ Posição inválida ou já ocupada!")
            input("Aperte Enter para continuar...") # Espera o usuário em vez de pausar por tempo
            continue

        # No tabuleiro, nós salvamos a versão COLORIDA do jogador
        tabuleiro[escolha] = cores[jogador_atual]
        jogadas += 1

        # Passamos a versão colorida para verificar a vitória corretamente no tabuleiro
        if verificar_vitoria(tabuleiro, cores[jogador_atual]):
            clear_output()
            exibir_tabuleiro(tabuleiro)
            print(f"🎉 PARABÉNS! O jogador {cores[jogador_atual]} venceu o jogo! 🏆\n")
            return

        # Alterna o jogador puro ("X" ou "O")
        jogador_atual = "O" if jogador_atual == "X" else "X"

    clear_output()
    exibir_tabuleiro(tabuleiro)
    print("👵 O jogo empatou! Deu velha!")

# Executa o jogo
jogo_da_velha()
